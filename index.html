async function checkTele() { try { // Thêm tham số timeout để bot phản hồi nhanh hơn const r = await fetch(`https://api.telegram.org/bot${TOKEN}/getUpdates?offset=${lastUpdateId + 1}&timeout=1`); const d = await r.json(); if (d.result && d.result.length > 0) { for (let u of d.result) { // Cập nhật ID tin nhắn ngay lập tức để tránh trùng lặp lastUpdateId = u.update_id; if (!u.message || !u.message.text) continue; const m = u.message.text.trim().toLowerCase(); console.log("Nhận lệnh:", m); // Kiểm tra trong Console của trình duyệt if (m === "/start" || m === "/ping") { const p = document.getElementById('goldPrice').innerText; const rsi = document.getElementById('rsiVal').innerText; const tr = marketTrend === "up" ? "TĂNG 📈" : "GIẢM 📉"; sendTele(`✅ BINGX V15.1 ONLINE\n💰 Giá: ${p}\n📈 RSI: ${rsi}\n📊 Trend: ${tr}\n🛡️ Trạng thái: ${isTrading ? "ĐANG CÓ LỆNH" : "ĐANG QUÉT"}`); continue; } // Các lệnh khác giữ nguyên... if (m.startsWith("/sync")) { let val = parseFloat(m.replace("/sync","").trim()); if(!isNaN(val)) { priceOffset = val - basePrice; sendTele(`🎯 Đã Sync giá BingX: ${val}`); } } if (m === "/done") { isTrading = false; document.getElementById('tradeUI').style.display="none"; sendTele("✅ Đã reset trạng thái."); } } } } catch (e) { console.error("Lỗi Telegram:", e); } }
