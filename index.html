function execute(type, p) { side = type; entry = p; isTrading = true; hitTPs = []; // 1. TỰ ĐỘNG XÁC ĐỊNH KHOẢNG CÁCH TP (GAP) // Nếu MarketTrend là UP/DOWN mạnh, dùng Gap lớn. Nếu Sideway dùng Gap nhỏ. let gap = 3.5; // Mặc định if (marketTrend !== "SIDEWAY") { gap = 6.0; // Nới xa TP khi có xu hướng mạnh (Trend) } else { gap = 2.5; // Chốt cực nhanh khi thị trường đi ngang (Sideway) } // 2. TÍNH TOÁN VÙNG ENTRY (Zone 5 giá) let entryStart = p; let entryEnd = (type === "BUY") ? p + 5 : p - 5; // 3. TÍNH SL (Trend nới rộng SL hơn để tránh quét) sl = (type === "BUY") ? entry - (marketTrend !== "SIDEWAY" ? 20 : 12) : entry + (marketTrend !== "SIDEWAY" ? 20 : 12); // 4. TẠO DANH SÁCH 6 MỨC TP ĐỘNG tps = Array.from({length:6}, (_,i) => parseFloat((type==="BUY" ? entry+(i+1)*gap : entry-(i+1)*gap).toFixed(2))); // Cập nhật giao diện document.getElementById('tradeUI').style.display = "block"; document.getElementById('sideVal').innerText = type + (marketTrend !== "SIDEWAY" ? " (TREND)" : " (SIDE)"); document.getElementById('enVal').innerText = type === "BUY" ? `${entryStart.toFixed(2)} - ${entryEnd.toFixed(2)}` : `${entryEnd.toFixed(2)} - ${entryStart.toFixed(2)}`; // Gửi tin nhắn Telegram let msg = `🚀 **BINGX: ${type} ${marketTrend !== "SIDEWAY" ? "🔥 TREND" : "💤 SIDEWAY"}**\n`; msg += `📍 Vùng Entry: **${entryStart.toFixed(2)} - ${entryEnd.toFixed(2)}**\n`; msg += `🛑 Stop Loss: ${sl.toFixed(2)}\n`; msg += `🎯 Mục tiêu (Gap ${gap}): ${tps[0]} -> ${tps[5]}\n`; msg += `📊 Trạng thái: ${marketTrend === "UP" ? "Tăng mạnh" : "Giảm mạnh"}`; sendTele(msg); }
